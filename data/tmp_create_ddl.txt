create table SMARTPHONE
(
    ID          SERIAL         primary key,
    CPU         VARCHAR(255)   not null,
    DESCRIPTION VARCHAR(255),
    GPU         VARCHAR(255)   not null,
    MODEL_NAME  VARCHAR(255)   not null,
    PRICE       DECIMAL(19, 2) not null,
    RAM         INTEGER        not null,
    ROM         INTEGER        not null,
    BRAND_ID    BIGINT         not null,
    constraint FK_SMARTPHONE_TO_BRAND
        foreign key (BRAND_ID) references BRAND (ID)
);

create table PURCHASE_ORDER
(
    ID         SERIAL         primary key,
    FULL_PRICE DECIMAL(19, 2) not null,
    NUMBER     BIGINT         not null,
    STATUS     VARCHAR(255)   not null,
    USER_ID    BIGINT         not null
);

create table ORDER_SMARTPHONES
(
    SMARTPHONE_ID BIGINT not null,
    ORDER_ID      BIGINT not null,
    constraint FK_ORDPHO_TO_SMARTPHONE
        foreign key (SMARTPHONE_ID) references SMARTPHONE (ID),
    constraint FK_ORDPHO_TO_PUR_ORDER
        foreign key (ORDER_ID) references PURCHASE_ORDER (ID)
